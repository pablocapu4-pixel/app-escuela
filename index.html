// --- FUNCIÓN ACTUALIZADA PARA CARGAR MATERIAS POR CUATRIMESTRE --- function cargarMenuMaterias() { const select = document.getElementById("select-materia"); const materiasRaiz = docenteActivo.MATERIAS || {}; select.innerHTML = ''; // 1. Entramos al Año (Ej: "AÑO 1") Object.keys(materiasRaiz).forEach(añoKey => { const cuatrimestres = materiasRaiz[añoKey]; // 2. Entramos al Cuatrimestre (Ej: "1er CUATRIMESTRE") Object.keys(cuatrimestres).forEach(cuatKey => { const listaMaterias = cuatrimestres[cuatKey]; if (Array.isArray(listaMaterias)) { listaMaterias.forEach(materia => { let opt = document.createElement("option"); // Extraemos el número de cuatrimestre del texto (Ej: "1er" -> "1") const numCuat = cuatKey.match(/\d+/) ? cuatKey.match(/\d+/)[0] : "1"; // Extraemos el número de año (Ej: "AÑO 1" -> "1") const numAño = añoKey.match(/\d+/) ? añoKey.match(/\d+/)[0] : "1"; // Guardamos todo en el value para usarlo después opt.value = `${materia.trim()}|${numAño}|${numCuat}`; opt.textContent = `${materia.trim()} (${numAño}° Año - ${numCuat}° Cuat.)`; select.appendChild(opt); }); } }); }); } // --- FUNCIÓN ACTUALIZADA PARA MOSTRAR ALUMNOS --- async function mostrarAlumnos() { const val = document.getElementById("select-materia").value; if(!val) return; const [materia, año, cuat] = val.split('|'); document.getElementById("titulo-materia").innerText = `${materia} (${cuat}° Cuatrimestre)`; const container = document.getElementById("lista-alumnos"); document.getElementById("planilla-card").classList.remove("hidden"); container.innerHTML = "Cargando alumnos..."; try { // Buscamos alumnos que coincidan con el número de año const snap = await db.collection("usuarios") .where("rol", "==", "estudiante") .where("año", "==", año) .get(); if (snap.empty) { container.innerHTML = `

No hay alumnos cargados en ${año}° año.

`; return; } container.innerHTML = ""; snap.forEach(doc => { const alu = doc.data(); container.innerHTML += `
${alu.nombre}
`; }); } catch (err) { container.innerHTML = "Error: " + err.message; } } // --- FUNCIÓN ACTUALIZADA PARA GUARDAR NOTAS --- async function guardarNotas() { const val = document.getElementById("select-materia").value; const [materia, año, cuat] = val.split('|'); const batch = db.batch(); const filas = document.querySelectorAll(".student-row"); filas.forEach(fila => { const dni = fila.dataset.dni; const rite = document.getElementById(`rite-${dni}`).value; const nota = parseFloat(document.getElementById(`nota-${dni}`).value); if(rite || !isNaN(nota)) { // El ID del documento ahora es más único: DNI + Materia + Cuatrimestre const idDoc = `${dni}_${materia.replace(/\s+/g, '')}_C${cuat}`; const ref = db.collection("evaluaciones").doc(idDoc); batch.set(ref, { alumnoDni: dni, materia: materia, año: año, cuat: cuat, rite: rite, nota: nota, fecha: new Date().toISOString() }); } }); await batch.commit(); alert(`Planilla de ${materia} (${cuat}° Cuat.) guardada con éxito.`); }
